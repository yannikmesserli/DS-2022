LineOfSightAnalysis

                  
                  const landmarks = [
                    {
                      name: "Neues Rathaus",
                      threshold: 5,
                      position: new Point({ /* ... */ }),
                    },
                    {
                      name: "St Peter's Church - Tower 1",
                      threshold: 15,
                      position: new Point({ /* ... */ }),
                    },
                    {
                      name: "St Peter's Church - Tower 2",
                      threshold: 15,
                      position: new Point({ /* ... */ }),
                    },
                  ];

                  // Make an analysis with a target for each of our landmarks.
                  const targets = landmarks.map(({ position }) => {
                    return new LineOfSightAnalysisTarget({ position })
                  });
                  const analysis = new LineOfSightAnalysis({ targets });

                  view.analyses.add(analysis);
                  
                
                  
                  view.on("pointer-move", async (e) => {   
                    // Cancellation logic omitted for brevity
                    
                    const { ground, results } = await view.hitTest(e);

                    analysis.observer = position
                      ? new LineOfSightAnalysisObserver({ position })
                      : null;
                  });
                  
                
                  
                  view.on("immediate-click", async () => {
                    // Cancellation logic omitted for brevity
              
                    const { results } = await view.whenAnalysisView(analysis);
              
                    const observer = analysis.observer;
                    const staticAnalysis = new LineOfSightAnalysis({ observer });
              
                    for (let i = 0; i < landmarks.length; ++i) {
                      const landmark = landmarks[i];
                      const intersection = results.getItemAt(i)?.intersectedLocation;
                      const target = analysis.targets.getItemAt(i);
              
                      if (!intersection || !target) {
                        continue;
                      }
              
                      if (target.position.distance(intersection) < landmark.threshold) {
                        staticAnalysis.targets.push(target);
                      }
                    }
              
                    view.analyses.add(staticAnalysis);
                  });

Exporting profile data

                  
                  let abortController = null;

                  const button = document.createElement("calcite-button");
                  button.textContent = "Download as CSV";
                  button.onclick = async () => {
                    const groundProfileLine = widget.profiles.getItemAt(0);
                    
                    // Abort any previous operations.
                    abortController?.abort();
                    const { signal } = (abortController = new AbortController());

                    try {
                      // Make sure the profile is fully generated/refined
                      await reactiveUtils.whenOnce(() => groundProfileLine.progress === 1, signal);
                      throwIfAborted(signal);
                    } catch (e) {
                      throwIfNotAbortError(e);
                    }
                    
                    // Generate CSV data
                    const fields = ["distance", "elevation", "x", "y", "z"];
                    const parser = new json2csv.Parser({ field });
                    const csv = parser.parse(groundProfileLine.samples);
                    
                    // Save file with FileSaver.js
                    const type "text/csv;charset=utf-8";
                    const blob = new Blob([csv], { type });
                    saveAs(blob, "profile-data.csv");
                  }

                  view.ui.add(button, "bottom-left");

  view = new SceneView({
    map: webscene,
    container: "viewDiv"
  });
      

  let layer, layerView;
  

  webscene.load().then(function() {
    layer = webscene.layers.find(
      (l) => l.title === "Helsinki_buildings"
    );
    view.whenLayerView(layer).then(function (lV) {
      layerView = lV;
    });
  });

Adding a Time Slider

  const start = new Date(1900, 0, 1);
  const end = new Date(2021, 0, 1);
  

  const timeSlider = new TimeSlider({
    container: "timeContainer",
    mode: "cumulative-from-start",
    fullTimeExtent: { start, end },
    values: [end],
    stops: {
      interval: {
        value: 5,
        unit: "years"
      },
      timeExtent: { start, end }
    }
  });
  timeSlider.watch("timeExtent", function(timeExtent) {
    const maxYear = timeExtent.end.getFullYear();
    

    layerView.filter = new FeatureFilter({
      where: `yearCompleted <= ${maxYear}`
    });
  });
