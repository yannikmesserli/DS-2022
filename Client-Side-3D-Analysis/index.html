Exporting profile data

                  
                  async function downloadCSV() {
                    const groundProfileLine = widget.profiles.getItemAt(0);

                    // Wait for the profile to be fully generated. Cancellation
                    // and error handling omitted for brevity
                    await reactiveUtils.whenOnce(() => groundProfileLine.progress === 1);
                    
                    exportToCSV(groundProfileLine.samples);
                  }
                  
                
                      
                    function exportToCSV(samples) {                
                      const fields = ["distance", "elevation", "x", "y", "z"];
                      const csv = new json2csv.Parser({ fields }).parse(samples);
                      
                      // Offer file for download with FileSaver.js
                      const blob = new Blob([csv], { type: "text/csv;charset=utf-8" });
                      saveAs(blob, "profile-data.csv");
                    }
