                        }
                      </style>

                      <script src="https://jsdev.arcgis.com/4.23/"></script>
                  
                      <script>
                        require(["esri/Map", "esri/views/SceneView"], (Map, SceneView) => {
                          const map = new Map({
                            basemap: "satellite",
                          });
                  
                          const view = new SceneView({
                            container: "viewDiv",
                            map: map,
                          });
                        });
                      </script>
                    </head>
                  
                    <body>
                      <div id="viewDiv"></div>
                    </body>
                  </html>
