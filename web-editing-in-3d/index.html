From editing in 2D to editing in 3D

  • Using the Editor widget in a SceneView works out of the box.
  • Using the SketchViewModel also works 3D.
  • FeatureForm and Editor widget is the same in 2D and 3D.
  • No need to prepare data specifically for 3D.

Placing trees on the ground

                  
                  const view = new SceneView(/* ... */);

                  const sketchViewModel = new SketchViewModel({
                    view,
                    defaultUpdateOptions: {
                      enableRotation: true,
                      enableScaling: true,
                    },
                  });

                  // Small widget where we will add a few buttons
                  const widget = document.createElement("div");
                  widget.className = "esri-widget";
                  widget.style.display = "flex";
                  widget.style.flexDirection = "column";
                  widget.style.gap = "10px";
                  widget.style.padding = "10px";
                  view.ui.add(widget, "top-right");
                  
                
                  
                  const trees = new GraphicsLayer({ 
                    elevationInfo: { mode: "on-the-ground" }
                  });
                  view.map.add(trees);

                  // Get a `PointSymbol3D` for our graphics.
                  const treeSymbol = await new WebStyleSymbol({
                    name: "Larix",
                    styleName: "EsriRealisticTreesStyle",
                  }).fetchSymbol();

                  const button = document.createElement("calcite-button");
                  button.textContent = "Add tree";

                  button.addEventListener("click", () => {
                    sketchViewModel.layer = trees;
                    sketchViewModel.pointSymbol = treeSymbol;
                    sketchViewModel.create("point");
                  });
                  
                
                  
                  const airplanes = new GraphicsLayer({ 
                    elevationInfo: { mode: "relative-to-ground" }
                    // elevationInfo: { mode: "absolute-height" }
                  });
                  view.map.add(airplanes);

                  // Get a `PointSymbol3D` for our graphics.
                  const airplaneSymbol = await new WebStyleSymbol({
                    name: "Airplane_Propeller",
                    styleName: "EsriRealisticTransportationStyle",
                  }).fetchSymbol();

                  const button = document.createElement("calcite-button");
                  button.textContent = "Add airplane";

                  button.addEventListener("click", () => {
                    sketchViewModel.layer = airplanes;
                    sketchViewModel.pointSymbol = airplaneSymbol;
                    sketchViewModel.create("point");
                  });
                  
                
                  
                  const antennas = new GraphicsLayer({ 
                    elevationInfo: { mode: "relative-to-scene" }
                  });
                  view.map.add(antennas);

                  // Get a `PointSymbol3D` for our graphics.
                  const antennaSymbol = await new WebStyleSymbol({
                    name: "Cell_Phone_Antenna",
                    styleName: "EsriInfrastructureStyle",
                  }).fetchSymbol();

                  const button = document.createElement("calcite-button");
                  button.textContent = "Add airplane";
                  
                  button.addEventListener("click", () => {
                    sketchViewModel.layer = antennas;
                    sketchViewModel.pointSymbol = antennaSymbol;
                    sketchViewModel.create("point");
                  });
                  
                
                  
                  const terminals = new GraphicsLayer({ 
                    elevationInfo: { mode: "relative-to-scene" }
                  });
                  view.map.add(terminals);

                  // Make sketching easier with snapping.
                  sketchViewModel.snappingOptions.enabled = true;
                  sketchViewModel.snappingOptions.featureSources = [{ layer: terminals }];

                  const terminalSymbol = new PolygonSymbol3D({
                    symbolLayers: [
                      new ExtrudeSymbol3DLayer({
                        size: 5,
                        material: { color: [255, 255, 255, 1] },
                      }),
                    ],
                  })

                  const button = document.createElement("calcite-button");
                  button.textContent = "Add terminal";
                  
                  button.addEventListener("click", () => {
                    sketchViewModel.layer = terminals;
                    sketchViewModel.polygonSymbol = terminalSymbol;
                    sketchViewModel.create("polygon");
                  });

FeatureLayer and Editor widget

                  
                  const pointLayer = new FeatureLayer({
                    title: "Airplanes",
                    url: "https://services.arcgis ...",
                    elevationInfo: { mode: "absolute-height" },
                    outFields: ["*"],
                    renderer: new UniqueValueRenderer({
                      field: "TYPE",
                      uniqueValueInfos: [
                        { value: "1", label: "Helicopter", symbol: helicopterSymbol },
                        { value: "2", label: "Airplane", symbol: airplaneSymbol },
                      ],
                      visualVariables: [
                        new SizeVariable({
                          axis: "height",
                          field: "SIZE",
                          valueUnit: "meters",
                        }),
                        new RotationVariable({
                          field: "ROTATION",
                          rotationType: "geographic",
                        }),
                      ],
                    }),
                  });

                  view.map.add(pointLayer);
                  
                
                                  
                  const formTemplate = new FormTemplate({
                    elements: [
                      new FieldElement({ fieldName: "ROTATION", label: "Rotation" }),
                      new FieldElement({ fieldName: "SIZE", label: "Size" }),
                    ],
                  });
                
                  const editor = new Editor({
                    view,
                    layerInfos: [{ layer: pointLayer, formTemplate }],
                  });
                
                  view.ui.add(editor, "top-right");
